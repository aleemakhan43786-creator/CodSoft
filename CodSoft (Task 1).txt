task_lst=[]

def add_task():
    enter_task=input("Enter task: ")
    task_lst.append(enter_task)

def remove_task():
    rem_task=input("Enter task to remove: ")
    if rem_task in task_lst:
        task_lst.remove(rem_task)
    else:
        print("Task doesn't exist")

def view_task():
    print(task_lst)


while True:
    print("Main Menu")
    print("1.Add Task")
    print("2.Remove Task")
    print("3.View Task")
    print("4.Exit")
    ch=int(input("Enter your choice: "))
    if ch==1:
        add_task()
    elif ch==2:
        remove_task()
    elif ch==3:
        view_task()
    elif ch==4:
        break
    else:
        print("Invalid choice,Try again")
